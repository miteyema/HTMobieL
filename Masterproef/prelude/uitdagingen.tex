\newcommand{\stringcases}[3]{
  \romannumeral

\newcommand{\str@case}[3]{
  \ifnum\pdf@strcmp{\unexpanded{#1}}{\unexpanded{#2}}=\z@
    \expandafter\@firstoftwo
  \else
    \expandafter\@secondoftwo
  \fi
    {\str@case@end{#3}}
    {\str@case{#1}}
}

\long\def\str@case@end#1#2\q@stop{\z@#1}
